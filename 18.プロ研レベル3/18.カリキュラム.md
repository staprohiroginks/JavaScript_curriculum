<h2><b>問題50</b></h2>
<p>「6～60 のランダムな値が出力されるプログラムを選択しなさい」<br>

function createRandom() {
    let num = Math.floor(Math.random());
    return num;
}

let ans = 0;
let arr = [1, 2, 3];
let randNum = createRandom();

for (let element = arr) {
    ans = ans + element * randNum;
}
